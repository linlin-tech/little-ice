4. 摘要触发与生成服务（services/summary_service.rs）
这是核心逻辑，严格按照 113 原则实现。

关键设计：

触发判断：摘要触发与生成服务中“触发判断”是在AI回复后，用多线程调用生成摘要。通过数据库查询最近消息和最新摘要的 last_message_id，计算轮数差。
摘要生成：调用 ai_service 的非流式接口（stream=false）进行摘要生成。
持久化：生成成功后，插入新的 ChatSummary。
use crate::errors::AppError;
use crate::models::summary::{ChatSummary, SummaryContext};
use crate::db::repository::summary_repo::SummaryRepo;
use crate::services::ai_service::AiService;

pub struct SummaryService {
    summary_repo: SummaryRepo,
    ai_service: AiService,
}

impl SummaryService {
    /// 检查并执行摘要生成（用户提交问题后，AI 回复前调用）
    /// 返回生成的摘要内容（如果触发生成），否则返回 None
    pub async fn maybe_summarize(
        &self,
        chat_id: &str,
        current_round_num: u32, // 当前即将发送的用户问题是第几轮（从1开始）
    ) -> Result<Option<ChatSummary>, AppError> {
        // 1. 获取当前对话的最新摘要
        let latest_summary = self.summary_repo
            .find_latest_by_chat_id(chat_id)
            .await?;
        
        // 2. 根据 113 原则判断是否应触发
        let should_trigger = match &latest_summary {
            Some(summary) => {
                // 获取 last_message_id 对应的消息序号
                let last_msg_seq = self.get_msg_sequence_number(chat_id, &summary.last_message_id).await?;
                // 条件：轮次差 >= 3（注意：current_round_num 是用户即将发送的问题的轮次）
                current_round_num.saturating_sub(last_msg_seq) >= 3
            },
            None => {
                // 首次触发：当前轮次 >= 4（即用户提交第4个问题后）
                current_round_num >= 4
            }
        };

        if !should_trigger {
            return Ok(None);
        }

        // 3. 收集最近3轮对话（不含当前这轮）
        let recent_messages = self.get_recent_round_messages(chat_id, current_round_num, 3).await?;
        
        // 4. 生成摘要
        let summary_content = self.generate_summary(&recent_messages).await?;
        
        // 5. 提取关键词（从 AI 返回中解析）
        let (content, keywords) = self.parse_summary_response(&summary_content);
        
        // 6. 获取最近3轮的最后一条消息 ID（作为 last_message_id）
        let last_msg_id = recent_messages.last()
            .ok_or(AppError::Internal("No messages found for summarization".into()))?
            .id.clone();
        
        // 7. 创建并持久化摘要
        let new_summary = ChatSummary {
            id: Uuid::new_v4().to_string(),
            chat_id: chat_id.to_string(),
            content,
            keywords,
            last_message_id: last_msg_id,
            created_at: chrono::Utc::now(),
        };
        
        // 8. 事务：删除旧摘要（最多保留1条），插入新摘要
        self.summary_repo.replace_for_chat(chat_id, &new_summary).await?;
        
        Ok(Some(new_summary))
    }

    /// 获取当前上下文（用于构建 Prompt）
    pub async fn get_context_for_chat(&self, chat_id: &str) -> Result<SummaryContext, AppError> {
        let latest = self.summary_repo.find_latest_by_chat_id(chat_id).await?;
        Ok(latest.map(|s| SummaryContext {
            summary: Some(s.content),
            keywords: Some(s.keywords),
            last_message_id: Some(s.last_message_id),
        }).unwrap_or_default())
    }

    // ... 内部辅助方法（get_msg_sequence_number, get_recent_round_messages, generate_summary, parse_summary_response）
}

// 在 SummaryService::generate_summary 中
// 使用 tokio::time::timeout 设置 5 秒超时
use tokio::time::timeout;

async fn call_summary_api(&self, messages: &[Message]) -> Result<String, AppError> {
    let result = timeout(Duration::from_secs(5), self.ai_service.summarize(messages)).await;
    match result {
        Ok(Ok(text)) => Ok(text),
        Ok(Err(e)) => Err(AppError::SummaryGenerationFailed(e.to_string())),
        Err(_) => {
            // 超时：使用截断 fallback
            Ok(fallback_summarize(messages))
        }
    }
}

8. 性能保障
数据库查询：仅查询 chat_summaries 单行记录，毫秒级。
摘要生成：非流式调用，预期 1-2 秒；超时降级至纯本地截断。
内存：仅缓存当前上下文（摘要 + 最近 1 轮），内存占用极低。
