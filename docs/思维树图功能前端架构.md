### 2.1 @xyflow/react 节点/边类型定义

```ts
// src/features/tree/types/xyflow.ts
import { Node, Edge } from '@xyflow/react';
import { TreeNode, Id } from '@/types/models';

// ===== XYFlow 自定义节点类型 =====
export type TreeNodeData = {
  label: string;
  treeNodeId: Id;
  roleId: Id;
  childCount: number;
  isRoot: boolean;
  isSelected: boolean;
  isExpanded: boolean;
};

export type TreeNodeFlowNode = Node<TreeNodeData, 'treeNode'>;
export type TreeNodeFlowEdge = Edge;

// ===== 转换函数 =====
export function treeNodeToFlowNode(
  tn: TreeNode,
  expandedIds: Set<Id>,
  selectedId: Id | null,
  depth: number
): TreeNodeFlowNode {
  return {
    id: tn.id,
    type: 'treeNode',
    position: { x: 0, y: 0 }, // 由 layout 算法计算
    data: {
      label: tn.title,
      treeNodeId: tn.id,
      roleId: tn.roleId,
      childCount: tn.childCount ?? 0,
      isRoot: tn.parentId === null,
      isSelected: tn.id === selectedId,
      isExpanded: expandedIds.has(tn.id),
    },
  };
}

export function treeNodesToFlowEdges(
  parentId: Id,
  children: TreeNode[]
): TreeNodeFlowEdge[] {
  return children.map((child) => ({
    id: `${parentId}->${child.id}`,
    source: parentId,
    target: child.id,
    type: 'smoothstep',
    animated: false,
  }));
}
```

### 2.2 布局算法（使用 dagre）

```ts
// src/features/tree/hooks/useTreeLayout.ts
import { useMemo } from 'react';
import dagre from 'dagre';
import { Node, Edge } from '@xyflow/react';
import { TreeNodeFlowNode } from '../types/xyflow';

const NODE_WIDTH = 200;
const NODE_HEIGHT = 60;
const RANK_SEP = 100;
const NODE_SEP = 50;

export function useTreeLayout(
  nodes: TreeNodeFlowNode[],
  edges: Edge[],
  direction: 'TB' | 'LR' = 'TB'
): { nodes: TreeNodeFlowNode[]; edges: Edge[] } {
  return useMemo(() => {
    const dagreGraph = new dagre.graphlib.Graph();
    dagreGraph.setDefaultEdgeLabel(() => ({}));
    dagreGraph.setGraph({
      rankdir: direction,
      ranksep: RANK_SEP,
      nodesep: NODE_SEP,
    });

    nodes.forEach((node) => {
      dagreGraph.setNode(node.id, {
        width: NODE_WIDTH,
        height: NODE_HEIGHT,
      });
    });

    edges.forEach((edge) => {
      dagreGraph.setEdge(edge.source, edge.target);
    });

    dagre.layout(dagreGraph);

    const layoutedNodes = nodes.map((node) => {
      const dagreNode = dagreGraph.node(node.id);
      return {
        ...node,
        position: {
          x: dagreNode.x - NODE_WIDTH / 2,
          y: dagreNode.y - NODE_HEIGHT / 2,
        },
      };
    });

    return { nodes: layoutedNodes, edges };
  }, [nodes, edges, direction]);
}
```

内部实现关键：

```ts
// ===== 将 TreeNode 数据同步为 XYFlow 数据 =====
function buildFlowData(
  allNodes: Map<Id, TreeNode>,
  expandedIds: Set<Id>,
  selectedId: Id | null
): { nodes: TreeNodeFlowNode[]; edges: TreeNodeFlowEdge[] } {
  const nodes: TreeNodeFlowNode[] = [];
  const edges: TreeNodeFlowEdge[] = [];
  
  // 递归构建
  function traverse(nodeId: Id, depth: number) {
    const node = allNodes.get(nodeId);
    if (!node) return;
    
    nodes.push(treeNodeToFlowNode(node, expandedIds, selectedId, depth));
    
    if (expandedIds.has(nodeId)) {
      // 查找子节点
      const children = Array.from(allNodes.values())
        .filter(n => n.parentId === nodeId)
        .sort((a, b) => a.order - b.order);
      
      for (const child of children) {
        edges.push({
          id: `${nodeId}->${child.id}`,
          source: nodeId,
          target: child.id,
          type: 'smoothstep',
          animated: false,
        } as TreeNodeFlowEdge);
        traverse(child.id, depth + 1);
      }
    }
  }
  
  // 从根节点开始
  const rootNodes = Array.from(allNodes.values())
    .filter(n => n.parentId === null)
    .sort((a, b) => a.order - b.order);
    
  for (const root of rootNodes) {
    traverse(root.id, 0);
  }
  
  return { nodes, edges };
}
```

### 5.3.2 Store 中的删除逻辑

```ts
// src/features/tree/store.ts 删除相关 action

deleteNode: async (nodeId: Id) => {
  const state = get();
  
  // 1. 检查是否存在删除对话框状态
  if (state.deleteDialog) {
    // 从对话框流程进入
    await get().confirmDelete();
    return;
  }
  
  // 2. 直接调用的情况（如键盘快捷键）
  const node = state.allNodes.get(nodeId);
  if (!node) return;
  
  // 打开删除确认对话框
  set({
    deleteDialog: {
      nodeId,
      nodeTitle: node.title,
      isRoot: node.parentId === null,
      childCount: node.childCount ?? 0,
    },
  });
},

confirmDelete: async () => {
  const state = get();
  const dialog = state.deleteDialog;
  if (!dialog) return;
  
  try {
    // 调用后端递归删除
    await tauri.deleteTreeNode(dialog.nodeId);
    
    // 3. 从缓存中移除
    const newAllNodes = new Map(state.allNodes);
    const nodesToRemove = collectDescendants(dialog.nodeId, newAllNodes);
    nodesToRemove.forEach(id => newAllNodes.delete(id));
    
    // 4. 清理选中状态
    const newSelectedId = state.selectedNodeId === dialog.nodeId 
      ? null 
      : state.selectedNodeId;
    
    // 5. 关闭对话框
    set({
      allNodes: newAllNodes,
      selectedNodeId: newSelectedId,
      deleteDialog: null,
    });
    
    // 6. 同步到 appStore
    if (dialog.isRoot && dialog.nodeId === state.rootNodeIds.find(id => id === dialog.nodeId)) {
      // 如果删除的是当前根节点，清空选中
      useAppStore.getState().selectChat(null);
    }
    
  } catch (error) {
    console.error('Failed to delete node:', error);
    throw error;
  }
},

// 辅助函数：收集所有子孙节点 id
function collectDescendants(nodeId: Id, allNodes: Map<Id, TreeNode>): Id[] {
  const result: Id[] = [nodeId];
  allNodes.forEach((node, id) => {
    if (node.parentId === nodeId) {
      result.push(...collectDescendants(id, allNodes));
    }
  });
  return result;
}
```

```ts
// src/features/tree/store.ts 创建逻辑

createNode: async (title: string, parentId: Id | null, roleId?: Id) => {
  try {
    // 1. 调用后端创建节点
    const newNode = await tauri.createTreeNode({
      title,
      parentId,
      roleId: roleId ?? DEFAULT_ROLE_ID,
    });
    
    // 2. 更新本地缓存
    const state = get();
    const newAllNodes = new Map(state.allNodes);
    newAllNodes.set(newNode.id, newNode);
    
    // 3. 如果是根节点，添加到根节点列表
    const newRootNodeIds = [...state.rootNodeIds];
    if (parentId === null) {
      newRootNodeIds.unshift(newNode.id);
    }
    
    // 4. 如果创建的是子节点，更新父节点的 childCount
    if (parentId) {
      const parent = newAllNodes.get(parentId);
      if (parent) {
        newAllNodes.set(parentId, {
          ...parent,
          childCount: (parent.childCount ?? 0) + 1,
        });
      }
      
      // 5. 展开父节点
      const newExpanded = new Set(state.expandedNodeIds);
      newExpanded.add(parentId);
      set({ expandedNodeIds: newExpanded });
    }
    
    // 6. 更新状态
    set({
      allNodes: newAllNodes,
      rootNodeIds: newRootNodeIds,
    });
    
    // 7. 重新构建 flow 数据
    const { nodes, edges } = buildFlowData(
      newAllNodes,
      get().expandedNodeIds,
      get().selectedNodeId,
    );
    set({ flowNodes: nodes, flowEdges: edges });
    
    return newNode;
  } catch (error) {
    console.error('Failed to create node:', error);
    throw error;
  }
},
```

```ts
  if (node) {
    set({
      editingNodeId: nodeId,
      editingTitle: node.title,
    });
  }
},

setEditingTitle: (title: string) => {
  set({ editingTitle: title });
},

submitEditing: async () => {
  const state = get();
  const { editingNodeId, editingTitle } = state;
  
  if (!editingNodeId || !editingTitle.trim()) {
    set({ editingNodeId: null });
    return;
  }
  
  try {
    // 1. 调用后端重命名
    await tauri.renameTreeNode(editingNodeId, editingTitle.trim());
    
    // 2. 更新本地缓存
    const newAllNodes = new Map(state.allNodes);
    const node = newAllNodes.get(editingNodeId);
    if (node) {
      newAllNodes.set(editingNodeId, {
        ...node,
        title: editingTitle.trim(),
        updatedAt: new Date().toISOString(),
      });
    }
    
    // 3. 重置编辑状态
    set({
      allNodes: newAllNodes,
      editingNodeId: null,
      editingTitle: '',
    });
    
    // 4. 重新构建 flow 数据
    const { nodes, edges } = buildFlowData(
      newAllNodes,
      get().expandedNodeIds,
      get().selectedNodeId,
    );
    set({ flowNodes: nodes, flowEdges: edges });
  } catch (error) {
    console.error('Failed to rename node:', error);
    throw error;
  }
},
```

### 5.7 键盘导航

```ts
// src/features/tree/hooks/useTreeKeyboardNav.ts
import { useCallback, useEffect } from 'react';
import { useTreeViewStore } from '../store';
import { useTreeExpansion } from './useTreeExpansion';

export function useTreeKeyboardNav(containerRef: React.RefObject<HTMLDivElement>) {
  const { allNodes, selectedNodeId, setSelectedNode, deleteNode } = useTreeViewStore();
  const { toggleExpand } = useTreeExpansion();
  
  const handleKeyDown = useCallback((e: KeyboardEvent) => {
    if (!selectedNodeId) return;
    
    switch (e.key) {
      case 'ArrowDown': {
        e.preventDefault();
        // 找到下一个兄弟节点或父节点
        const node = allNodes.get(selectedNodeId);
        if (node) {
          const siblings = Array.from(allNodes.values())
            .filter(n => n.parentId === node.parentId)
            .sort((a, b) => a.order - b.order);
          const currentIndex = siblings.findIndex(n => n.id === selectedNodeId);
          if (currentIndex < siblings.length - 1) {
            setSelectedNode(siblings[currentIndex + 1].id);
          } else if (node.parentId) {
            // 没有兄弟节点时，选择父节点
            setSelectedNode(node.parentId);
          }
        }
        break;
      }
      
      case 'ArrowUp': {
        e.preventDefault();
        const node = allNodes.get(selectedNodeId);
        if (node) {
          const siblings = Array.from(allNodes.values())
            .filter(n => n.parentId === node.parentId)
            .sort((a, b) => a.order - b.order);
          const currentIndex = siblings.findIndex(n => n.id === selectedNodeId);
          if (currentIndex > 0) {
            setSelectedNode(siblings[currentIndex - 1].id);
          }
        }
        break;
      }
      
      case 'ArrowRight': {
        e.preventDefault();
        toggleExpand(selectedNodeId);
        break;
      }
      
      case 'ArrowLeft': {
        e.preventDefault();
        toggleExpand(selectedNodeId);
        break;
      }
      
      case 'Delete':
      case 'Backspace': {
        e.preventDefault();
        const node = allNodes.get(selectedNodeId);
        if (node) {
          deleteNode(selectedNodeId);
        }
        break;
      }
    }
  }, [selectedNodeId, allNodes, setSelectedNode, toggleExpand, deleteNode]);
  
  useEffect(() => {
    const container = containerRef.current;
    if (!container) return;
    
    container.addEventListener('keydown', handleKeyDown);
    container.setAttribute('tabindex', '0');
    
    return () => {
      container.removeEventListener('keydown', handleKeyDown);
    };
  }, [handleKeyDown, containerRef]);
}
```

```ts
// 使用 LRU 缓存控制节点展开数量
const MAX_EXPANDED_NODES = 100;

toggleExpand: (nodeId: Id) => {
  const state = get();
  const newExpanded = new Set(state.expandedNodeIds);
  
  if (newExpanded.has(nodeId)) {

      get().loadChildren(nodeId);
    }
  }
  
  set({ expandedNodeIds: newExpanded });
},
```
